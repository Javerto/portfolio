Python ile Finansal Veri Analizi

Pandas ve NumPy Kullanarak Temel Analiz

Python finansal analiz için mükemmel bir araç. İşte basit bir hisse senedi analizi örneği:

import pandas as pd
import numpy as np
import yfinance as yf
import matplotlib.pyplot as plt

# Hisse senedi verilerini çekme
def get_stock_data(ticker, period="1y"):
    stock = yf.Ticker(ticker)
    data = stock.history(period=period)
    return data

# Temel teknik analiz
def technical_analysis(data):
    # Basit Hareketli Ortalama
    data['SMA_20'] = data['Close'].rolling(window=20).mean()
    data['SMA_50'] = data['Close'].rolling(window=50).mean()
    
    # RSI Hesaplama
    delta = data['Close'].diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=14).mean()
    rs = gain / loss
    data['RSI'] = 100 - (100 / (1 + rs))
    
    return data

# Örnek kullanım
if __name__ == "__main__":
    # Apple hisse senedi verileri
    aapl_data = get_stock_data("AAPL")
    analyzed_data = technical_analysis(aapl_data)
    
    print("Son 5 günlük veriler:")
    print(analyzed_data.tail())
    
    # Grafik çizimi
    plt.figure(figsize=(12, 6))
    plt.plot(analyzed_data.index, analyzed_data['Close'], label='Kapanış Fiyatı')
    plt.plot(analyzed_data.index, analyzed_data['SMA_20'], label='20 Günlük SMA')
    plt.plot(analyzed_data.index, analyzed_data['SMA_50'], label='50 Günlük SMA')
    plt.legend()
    plt.title('AAPL Hisse Senedi Analizi')
    plt.show()

SQL ile Finansal Veri Sorgulama

Window Functions ile Gelişmiş Analiz

SQL'de pencere fonksiyonları finansal analiz için çok güçlüdür. İşte örnek sorgular:

-- Portföy performans analizi
WITH portfolio_performance AS (
    SELECT 
        stock_symbol,
        trade_date,
        close_price,
        LAG(close_price) OVER (PARTITION BY stock_symbol ORDER BY trade_date) as prev_close,
        -- Günlük getiri
        (close_price - LAG(close_price) OVER (PARTITION BY stock_symbol ORDER BY trade_DATE)) / 
        LAG(close_price) OVER (PARTITION BY stock_symbol ORDER BY trade_DATE) * 100 as daily_return
    FROM stock_prices
    WHERE trade_date >= CURRENT_DATE - INTERVAL '90 days'
),
performance_summary AS (
    SELECT 
        stock_symbol,
        AVG(daily_return) as avg_daily_return,
        STDDEV(daily_return) as volatility,
        COUNT(*) as trading_days
    FROM portfolio_performance
    WHERE daily_return IS NOT NULL
    GROUP BY stock_symbol
)
SELECT 
    stock_symbol,
    avg_daily_return,
    volatility,
    -- Sharpe Oranı (basitleştirilmiş)
    avg_daily_return / NULLIF(volatility, 0) as sharpe_ratio
FROM performance_summary
ORDER BY sharpe_ratio DESC;

Modern ETL İşlem Hatları

Python ile Basit ETL Pipeline

İşte finansal veriler için basit bir ETL pipeline örneği:

import pandas as pd
import requests
import sqlite3
from datetime import datetime

class FinancialETL:
    def __init__(self, api_key=None):
        self.api_key = api_key
        self.raw_data = None
        self.processed_data = None
    
    def extract(self, source_type='api', **kwargs):
        """Veri çekme işlemi"""
        if source_type == 'api':
            # Örnek API çağrısı
            url = f"https://api.example.com/financial-data?apikey={self.api_key}"
            response = requests.get(url)
            self.raw_data = response.json()
        elif source_type == 'file':
            # Dosyadan okuma
            file_path = kwargs.get('file_path')
            self.raw_data = pd.read_csv(file_path)
        
        return self.raw_data
    
    def transform(self):
        """Veri dönüşümü"""
        if isinstance(self.raw_data, dict):
            # JSON verisini DataFrame'e dönüştür
            df = pd.DataFrame(self.raw_data['data'])
        else:
            df = self.raw_data.copy()
        
        # Temizleme ve dönüşüm işlemleri
        df['date'] = pd.to_datetime(df['date'])
        df['price'] = pd.to_numeric(df['price'], errors='coerce')
        df['volume'] = pd.to_numeric(df['volume'], errors='coerce')
        
        # Hesaplanmış alanlar
        df['price_change'] = df['price'].pct_change()
        df['volume_ma'] = df['volume'].rolling(window=5).mean()
        
        self.processed_data = df.dropna()
        return self.processed_data
    
    def load(self, target='database', **kwargs):
        """Veri yükleme"""
        if target == 'database':
            conn = sqlite3.connect(kwargs.get('db_path', 'financial_data.db'))
            self.processed_data.to_sql('financial_data', conn, if_exists='append', index=False)
            conn.close()
        elif target == 'file':
            self.processed_data.to_csv(kwargs.get('output_path'), index=False)
    
    def run_pipeline(self, source_type='api', target='database', **kwargs):
        """Tam ETL pipeline'ını çalıştır"""
        print("Veri çekme başlıyor...")
        self.extract(source_type, **kwargs)
        
        print("Veri dönüşümü başlıyor...")
        self.transform()
        
        print("Veri yükleme başlıyor...")
        self.load(target, **kwargs)
        
        print("ETL işlemi tamamlandı!")

# Kullanım örneği
if __name__ == "__main__":
    etl = FinancialETL(api_key="your_api_key_here")
    etl.run_pipeline(source_type='file', file_path='stock_data.csv', target='database', db_path='financial.db')
